128  void setChannelData(double _set_point, double _sensor_reading, double PID_out_DC, int _state)
129  {
130  this->set_point = _set_point;
131  this->sensor_reading = _sensor_reading;
132 
133  if (filter_enabled)
134  {
135  if (this->reading_vec.size() < filter_size) {
136  this->reading_vec.push_back(_sensor_reading);
137  }
138  else {
139  this->reading_vec.erase(this->reading_vec.begin());
140  while (this->reading_vec.size() > filter_size) {
141  this->reading_vec.erase(this->reading_vec.begin());
142  }
143  this->reading_vec.push_back(_sensor_reading);
144  }
145 
146  double sum = std::accumulate(this->reading_vec.begin(), this->reading_vec.end(), 0.0);
147  double mean = sum / this->reading_vec.size();
148  this->filtered_sensor_reading = mean;
149  this->sensor_reading = mean;
150 
151  }
152  else {
153  this->filtered_sensor_reading = _sensor_reading;
154  }
155 
156  this->PID_out_DC = PID_out_DC;
157  this->state = _state;
158 
159  }
