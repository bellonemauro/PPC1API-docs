212  double movingAveragefilter(vector<double> &_reading_vec, double _sensor_reading)
213  {
214  if (_reading_vec.size() < m_filter_size) {
215  _reading_vec.push_back(_sensor_reading);
216  }
217  else {
218  _reading_vec.erase(_reading_vec.begin());
219  while (_reading_vec.size() > m_filter_size) {
220  _reading_vec.erase(_reading_vec.begin());
221  }
222  this->m_reading_vec.push_back(_sensor_reading);
223  }
224 
225  double sum = std::accumulate(_reading_vec.begin(), _reading_vec.end(), 0.0);
226  double filtered_reading = sum / _reading_vec.size();
227  return filtered_reading; //TODO: this is now the same, is filtered_sensor_reading really necessary ?
228  }
