214  double movingAveragefilter(vector<double> &_reading_vec, double _sensor_reading)
215  {
216  if (_reading_vec.size() < m_filter_size) {
217  _reading_vec.push_back(_sensor_reading);
218  }
219  else {
220  _reading_vec.erase(_reading_vec.begin());
221  while (_reading_vec.size() > m_filter_size) {
222  _reading_vec.erase(_reading_vec.begin());
223  }
224  this->m_reading_vec.push_back(_sensor_reading);
225  }
226 
227  double sum = std::accumulate(_reading_vec.begin(), _reading_vec.end(), 0.0);
228  double filtered_reading = sum / _reading_vec.size();
229  return filtered_reading; //TODO: this is now the same, is filtered_sensor_reading really necessary ?
230  }

665

668  if (this->instruction > instructions::loop) return false;
